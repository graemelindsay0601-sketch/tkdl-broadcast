// Helper: get a player's unlocked achievement metadata objects
function getPlayerUnlocks(player) {
  if(!leagueData) return [];
  const unlocks = (leagueData.playerAchievements || []).filter(u =>
    String(u.playerId||'').toLowerCase() === String(player.name||'').toLowerCase() ||
    u.playerId === player.playerId
  );
  const meta = leagueData.allAchievements || leagueData.achievementMetadata || [];
  return unlocks.map(u => meta.find(m => m.id === u.achievementId)).filter(Boolean);
}

function showSection(id){

  document
    .querySelectorAll(".app-section")
    .forEach(section => {

      section.classList.add(
        "hidden-section"
      );

    });

  document
    .getElementById(id)
    .classList.remove(
      "hidden-section"
    );

  document
    .querySelectorAll(".nav-button")
    .forEach(button => {

      button.classList.remove(
        "active-nav"
      );

    });

  document
    .querySelector(
      `[data-section="${id}"]`
    )
    .classList.add(
      "active-nav"
    );

}

function renderSeasonClock(){

  const now = new Date();

  const year =
    now.getFullYear();

  const month =
    now.getMonth();

  const monthNames = [

    "JANUARY",
    "FEBRUARY",
    "MARCH",
    "APRIL",
    "MAY",
    "JUNE",
    "JULY",
    "AUGUST",
    "SEPTEMBER",
    "OCTOBER",
    "NOVEMBER",
    "DECEMBER"

  ];

  const end =
    new Date(year, month + 1, 0);

  const totalDays =
    end.getDate();

  const currentDay =
    now.getDate();

  const daysLeft =
    totalDays - currentDay;

  const percent =
    Math.round(
      (currentDay / totalDays) * 100
    );

  document.getElementById(
    "seasonName"
  ).innerHTML = `

    ${monthNames[month]}
    ${year}

  `;

  document.getElementById(
    "seasonDaysLeft"
  ).innerHTML = `

    ${daysLeft} DAYS LEFT

  `;

  document.getElementById(
    "seasonPercent"
  ).innerHTML = `

    ${percent}% COMPLETE

  `;

  document.getElementById(
    "seasonProgressFill"
  ).style.width =
    `${percent}%`;

}

function getEloChange(stake, isWinner){

  const base =
    Number(stake || 1) * 2;

  return isWinner
    ? `+${base}`
    : `-${base}`;

}

function validateStakeInput(el) {
  const val = parseInt(el.value, 10);
  const hint = document.getElementById('sf-stake-hint');
  const p1 = document.getElementById('sf-player1')?.value;
  const p2 = document.getElementById('sf-player2')?.value;
  const players = leagueData ? leagueData.players || [] : [];
  let maxStake = Infinity;
  if (p1) { const pd = players.find(p => p.name === p1); if (pd) maxStake = Math.min(maxStake, pd.points || 0); }
  if (p2) { const pd = players.find(p => p.name === p2); if (pd) maxStake = Math.min(maxStake, pd.points || 0); }
  if (maxStake < Infinity) {
    el.max = maxStake;
    if (hint) hint.textContent = 'Max stake: ' + maxStake + ' points (lowest of the two players)';
  }
  if (val > maxStake) el.value = maxStake;
  if (val < 1 && el.value !== '') el.value = 1;
}

async function submitMatchNative() {
  const p1       = document.getElementById('sf-player1')?.value;
  const p2       = document.getElementById('sf-player2')?.value;
  const winner   = document.getElementById('sf-winner')?.value;
  const stake    = document.getElementById('sf-stake')?.value    || '3';
  const gameType = document.getElementById('sf-gametype')?.value || 'League';
  const errDiv   = document.getElementById('sf-error');
  const okDiv    = document.getElementById('sf-success');
  const btn      = document.getElementById('sf-submit-btn');
  errDiv.style.display = 'none';
  okDiv.style.display  = 'none';
  if (!p1 || !p2)                     { _sfErr(errDiv, 'Please select both players.');            return; }
  if (p1 === p2)                      { _sfErr(errDiv, 'Players must be different.');             return; }
  if (!winner)                        { _sfErr(errDiv, 'Please select a winner.');                return; }
  if (winner !== p1 && winner !== p2) { _sfErr(errDiv, 'Winner must be one of the two players.'); return; }
  btn.disabled    = true;
  btn.textContent = '⏳ Submitting...';

  // Stake validation
  const stakeNum = Number(stake);
  if (!stake || isNaN(stakeNum) || stakeNum < 1) { _sfErr(errDiv, 'Please enter a valid stake (minimum 1 point).'); return; }
  if (!Number.isInteger(stakeNum))               { _sfErr(errDiv, 'Stake must be a whole number.');                   return; }
  const players = leagueData ? leagueData.players || [] : [];
  const p1Data  = players.find(p => p.name === p1);
  const p2Data  = players.find(p => p.name === p2);
  if (p1Data && stakeNum > (p1Data.points || 0)) { _sfErr(errDiv, p1 + ' only has ' + (p1Data.points||0) + ' points — can\'t stake ' + stakeNum + '.'); return; }
  if (p2Data && stakeNum > (p2Data.points || 0)) { _sfErr(errDiv, p2 + ' only has ' + (p2Data.points||0) + ' points — can\'t stake ' + stakeNum + '.'); return; }
  // Game type validation
  if (!gameType || !gameType.trim()) { _sfErr(errDiv, 'Please enter a game type.'); return; }

  try {
    const baseUrl = API_URL.split('?')[0];
    const resp = await fetch(baseUrl + '?function=submitMatch', {
      method:  'POST',
      headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
      body:    new URLSearchParams({ function:'submitMatch', player1:p1, player2:p2, winner, stake, gameType })
    });
    const result = await resp.json();
    if (result.success) {
      okDiv.textContent   = '✅ Match submitted! ' + winner + ' wins.';
      okDiv.style.display = 'block';
      btn.textContent     = '✅ Submitted!';
      setTimeout(async () => {
        document.getElementById('sf-player1').value = '';
        document.getElementById('sf-player2').value = '';
        document.getElementById('sf-winner').innerHTML = '<option value="">— Select players first —</option>';
        okDiv.style.display = 'none';
        btn.disabled        = false;
        btn.textContent     = '⚔️ SUBMIT MATCH';
        try {
          const fresh = await fetch(API_URL);
          leagueData  = await fresh.json();
          renderHome(); renderLeaderboard(); renderPlayers();
        } catch(e) {}
      }, 2500);
    } else {
      _sfErr(errDiv, result.error || 'Submission failed. Please try again.');
      btn.disabled    = false;
      btn.textContent = '⚔️ SUBMIT MATCH';
    }
  } catch(e) {
    _sfErr(errDiv, 'Network error. Check your connection and try again.');
    btn.disabled    = false;
    btn.textContent = '⚔️ SUBMIT MATCH';
  }
}

/* =========================
   ACHIEVEMENT SCORE
========================= */

function getAchievementScore(
  achievementIds
){

  const metadata =
leagueData
  .allAchievements || [];

  let score = 0;

  achievementIds.forEach(id => {

    const achievement =
      metadata.find(
        a => a.id === id
      );

    if(!achievement)
      return;

    switch(
      String(
        achievement.rarity || ''
      ).toLowerCase()
    ){

      case 'common':
        score += 5;
        break;

      case 'rare':
        score += 15;
        break;

      case 'epic':
        score += 35;
        break;

      case 'legendary':
        score += 75;
        break;

      default:
        score += 5;

    }

  });

  return score;

}

function openAchievementsModal(playerName){

  if(
    !leagueData ||
    !leagueData.players
  ){
    return;
  }

  const player =
    leagueData.players.find(
      p => p.name === playerName
    );

  if(!player){
    return;
  }

  // Build unlocked achievements from leagueData.playerAchievements
  const allUnlocks = leagueData.playerAchievements || [];
  const allMeta    = leagueData.achievementMetadata || [];
  const playerUnlocks = allUnlocks.filter(u =>
    String(u.playerId||'').toLowerCase() === String(player.name||'').toLowerCase() ||
    u.playerId === player.playerId
  );
  const unlockedAchievements = playerUnlocks.map(u =>
    allMeta.find(meta => meta.id === u.achievementId)
  ).filter(Boolean);
  const unlockedIds = new Set(unlockedAchievements.map(a => a.id));

  const allAchievements =
    leagueData.achievementMetadata || [];

  const unlockedCount =
    unlockedIds.size;

  const totalCount =
    allAchievements.filter(
      a => String(a.hidden).toUpperCase() !== 'TRUE'
    ).length;

  document
    .getElementById("player-modal")
    .innerHTML = `

    <div
      class="modal-backdrop"
      onclick="closePlayerProfile()">

      <div
        class="player-modal achievement-modal"
        onclick="event.stopPropagation()">

        <button
          class="close-modal"
          onclick="closePlayerProfile()">

          ✕

        </button>

        <div class="achievement-header">

          <h2>🏆 ${playerName} — Achievement Cabinet</h2>

          <div class="achievement-progress">
            ${unlockedCount} / ${totalCount} Unlocked
          </div>

        </div>

        <div class="achievement-filter-bar">

          <button
            class="achievement-filter active-filter"
            onclick="filterAchievements('all', this)">
            All
          </button>

          <button
            class="achievement-filter"
            onclick="filterAchievements('unlocked', this)">
            Unlocked
          </button>

          <button
            class="achievement-filter"
            onclick="filterAchievements('locked', this)">
            Locked
          </button>

          <button
            class="achievement-filter"
            onclick="filterAchievements('legendary', this)">
            Legendary+
          </button>

        </div>

        <div class="achievement-full-grid">

          ${allAchievements.map(a => {

            const unlocked =
              unlockedIds.has(a.id);

            const isHidden = String(a.hidden).toUpperCase() === 'TRUE';
            
            const rarityClass =
              String(a.rarity || "common")
                .toLowerCase();

            const showDetails =
              unlocked || !isHidden;

            return `

              <div class="
                achievement-card
                achievement-filter-item
                rarity-${rarityClass}
                ${unlocked
                  ? "achievement-unlocked"
                  : "achievement-locked"}

                  ${unlocked

                    : (isHidden ? "❔" : (a.icon || "🔒"))}

                  <div class="achievement-name">
                    ${showDetails
                      ? (a.name || "Unknown")

/* =========================================================
   ACHIEVEMENT FILTERS
========================================================= */

function filterAchievements(
  filter,
  button
){

  document
    .querySelectorAll(
      '.achievement-filter'
    )
    .forEach(btn => {

      btn.classList.remove(
        'active-filter'
      );

    });

  button.classList.add(
    'active-filter'
  );

  document
    .querySelectorAll(
      '.achievement-filter-item'
    )
    .forEach(card => {

      let visible = false;

      if(filter === 'all'){
        visible = true;
      }

      if(
        filter === 'unlocked' &&
        card.classList.contains(
          'achievement-unlocked'
        )
      ){
        visible = true;
      }

      if(
        filter === 'locked' &&
        card.classList.contains(
          'achievement-locked'
        )
      ){
        visible = true;
      }

      if(
        filter === 'legendary' &&
        card.classList.contains(
          'rarity-legendary'
        )
      ){
        visible = true;
      }

      card.style.display =
        visible
          ? 'flex'
          : 'none';

    });

}

function animateCountUps(){

  const counters =
    document.querySelectorAll('.count-up');

  counters.forEach(counter => {

    const target =
      Number(
        counter.textContent
          .replace(/[^0-9]/g,'')
      );

    if(isNaN(target)) return;

    let current = 0;

    const increment =
      Math.max(
        1,
        Math.ceil(target / 40)
      );

    counter.textContent = '0';

    const timer = setInterval(() => {

      current += increment;

      if(current >= target){

        counter.textContent = target;

        clearInterval(timer);

      }else{

        counter.textContent = current;

      }

    }, 18);

  });

}
